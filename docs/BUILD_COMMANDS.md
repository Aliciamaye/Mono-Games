# Local-First Architecture Build Scripts

# Build for Web (Vercel - with PWA)
npm run build:web

# Build for Electron (Local-First - NO PWA)
npm run build:electron

# Build for Mobile/APK (Local-First - NO PWA)
npm run build:mobile

# After building, package Electron
npm run package:electron

# After building, package Android APK
npm run package:android
